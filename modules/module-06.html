# 1) Least privilege per tool — narrow scope, prefer read-only, split high-impact tools out
tools = [
  ReadOnly("search_docs",  scope="kb:read"),          # cannot write / send / delete
  Scoped(  "send_email",   scope="mail:send",           # NOT mail:read, NOT admin
           allow_to=["@yourcompany.com"]),          # allow-list recipients
]
# anti-pattern:  one broad token (repo+admin+mail) shared by every tool → huge blast radius

# 2) Authorization enforced OUTSIDE the model — the app checks, not the prompt
def call_tool(user, tool, args):
    if not policy.allows(user, tool, args):     # real RBAC on the REAL user, not the LLM's claim
        raise PermissionDenied
    if tool.irreversible:                     # delete / pay / deploy / send-external
        require_human_approval(user, tool, args) # human-in-the-loop gate
    return tool.run(args)
# the model can REQUEST anything; it can only DO what the code independently permits

# 3) Dual-LLM — a quarantined model reads untrusted content but cannot call tools (Willison)
summary = quarantined_llm(untrusted_email)   # may READ; NO tool access
if privileged_llm.decide(summary):          # sees only sanitized text, never raw attacker input
    send_email(...)                          # actions gated behind the trusted model

# 4) Treat every tool output as untrusted input — it can carry an injection
result = tool.run(args)                       # email body, web page, API JSON, another agent's msg
context.append(as_data(result))               # tag as DATA, never as an instruction
# allow-list tools + args; log EVERY tool call with the user, args, and outcome
audit.log(user, tool, args, result)
