# Normalize + strip invisible characters BEFORE any keyword filter
import unicodedata, re

def canonicalize(s):
    s = unicodedata.normalize("NFKC", s)          # fold full-width / homoglyph forms
    s = "".join(c for c in s
                if unicodedata.category(c) != "Cf")  # drop zero-width / Tag / BiDi (format) chars
    s = re.sub(r"\s+", " ", s)               # collapse spacing tricks: "i g n o r e" → "ignore"
    return s
