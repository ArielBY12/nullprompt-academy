# 1) Keep secrets server-side — NEVER in the system prompt or client
key = os.environ["MODEL_API_KEY"]         # server env, never shipped to the browser
system = "You are Acme SupportBot. Be concise."  # no keys, no URLs, no rules-as-secrets
# anti-pattern:  system = f"...internal token is {BUILD_TOKEN}..."   # leaks on extraction

# 2) Per-user authorization on every conversation id (stops IDOR)
convo = db.get_conversation(convo_id)
if convo.owner_id != current_user.id:      # enforce ownership server-side
    return http_403()                     # never trust the id from the client alone

# 3) Rate-limit + cap tokens per user (caps the LLM10 blast radius)
if ratelimit.exceeded(current_user.id):    # N requests / window, enforced at the edge
    return http_429()
req["max_tokens"] = min(req.get("max_tokens", 512), 1024)  # clamp — ignore client over-asks
DEBUG = False                            # disable debug/verbose flags in prod

# 4) Redact secret formats from output, tripwire the system prompt
CANARY = "NP-CANARY-9f31"              # unguessable string embedded in the system prompt
out = model(prompt)
if CANARY in out or SECRET_RE.search(out): # canary appears ⇒ prompt is leaking
    alert(); out = SECRET_RE.sub("[redacted]", out)
