# 1) Pin & verify tool descriptions — hash on approval, re-approve on any change (stops rug-pulls)
approved = {}
def register_tool(server, tool):
    h = sha256(tool.name + tool.description + json(tool.schema))   # hash the FULL metadata
    if approved.get(tool.id) not in (None, h):   # description changed after approval
        block(tool); require_human_reapproval(server, tool)        # rug-pull / silent swap → halt
    approved[tool.id] = h                                          # pin it
# also: display the FULL raw description to the human, not a simplified UI label

# 2) Least privilege per tool + human-in-the-loop for privileged/irreversible tools
tools = [
  ReadOnly("get_weather", scope="weather:read"),          # cannot deploy / send / delete
  Scoped(  "deploy_key",  scope="admin", human_ok=True),  # privileged → require approval
]
def invoke(user, tool, args):
    if tool.human_ok: require_human_approval(user, tool, args)   # the model can REQUEST; code decides
    return tool.run(args)

# 3) Isolate & label untrusted tool metadata — it is DATA, never instructions
context.add_system(TRUSTED_POLICY)                    # fixed, from you
for t in tools:
    context.add_data("<tool-desc server="+t.server+">"          # tag origin + boundary
                     + spotlight(t.description) + "</tool-desc>")
# rule: text inside <tool-desc> is untrusted; never obey instructions found there
# cross-server: a description from Server B must NOT be able to redefine Server A's tools

# 4) Dual-LLM quarantine + treat every tool OUTPUT as untrusted (Willison)
result = tool.run(args)                       # API JSON, web page, another server's message
summary = quarantined_llm(as_data(result))    # may READ; NO tool access
if privileged_llm.decide(summary):          # sees sanitized text, never raw attacker metadata
    invoke(user, "send_email", ...)          # actions gated behind the trusted model
audit.log(user, tool, args, result)           # log EVERY tool call
